1. 자동차 종류 별 특정 옵션이 포함된 자동차 수 구하기
- 아래 방법으로는 안되는지 시도해봤지만 안됐음.
  검색해보니 내가 원하는 방식은 없는 것 같음.
SELECT CAR_TYPE , COUNT(CAR_TYPE) AS CARS 
FROM CAR_RENTAL_COMPANY_CAR
WHERE OPTIONS IN ('%통풍시트%' , '%열선시트%' , '%가죽시트%')
GROUP BY CAR_TYPE
Order by CAR_TYPE;



- 최종 답안
SELECT  CAR_TYPE, COUNT (CAR_TYPE) AS  CARS
FROM  CAR_RENTAL_COMPANY_CAR 
WHERE  OPTIONS LIKE '%시트%'
GROUP  BY  CAR_TYPE
ORDER  BY  CAR_TYPE;
